(function immediateInvoke() {
        let wording = document.getElementById('hidden_text').innerHTML,
            counter = 0,
            typewriter = document.getElementById('text'),
            startCount = setInterval(typing, 200);

        function typing() {
            let stringSplit = wording.substr(0, counter);
            typewriter.innerHTML = stringSplit;
            counter++;
            if (counter > wording.length) {
                clearInterval(startCount);
            }
        }

    }());
